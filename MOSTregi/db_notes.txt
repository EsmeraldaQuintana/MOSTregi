# INIT DATABASTE:
$ sudo su - postgres
postgres@yakuza:~$ psql
postgres=# CREATE DATABASE most;
postgres=# CREATE USER pinotnoir WITH PASSWORD 'pinotnoir';
postgres=# ALTER ROLE pinotnoir SET client_encoding TO 'utf8';
postgres=# ALTER ROLE pinotnoir SET default_transaction_isolation TO 'read committed';
postgres=# ALTER ROLE pinotnoir SET timezone TO 'EST';
postgres=# GRANT ALL PRIVILEGES ON DATABASE most TO pinotnoir;
postgres=# ALTER ROLE pinotnoir CREATEROLE CREATEDB;
postgres=# \q
postgres@yakuza:~$ exit

# CREATE SUPERUSER
