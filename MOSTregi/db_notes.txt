# INIT DATABASTE:
$ sudo su - postgres
postgres@yakuza:~$ psql
postgres=# CREATE DATABASE mostregi;
postgres=# CREATE USER admin WITH PASSWORD 'admin';
postgres=# ALTER ROLE admin SET client_encoding TO 'utf8';
postgres=# ALTER ROLE admin SET default_transaction_isolation TO 'read committed';
postgres=# ALTER ROLE admin SET timezone TO 'EST';
postgres=# GRANT ALL PRIVILEGES ON DATABASE mostregi TO admin;
postgres=# \q
postgres@yakuza:~$ exit
